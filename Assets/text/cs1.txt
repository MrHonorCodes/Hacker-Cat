using System;
class Program{
    static void Main(){
        int[,] array2D = {
            { 1, 2, 3 },
            { 4, 5, 6 },
            { 7, 8, 9 }
        };
        int rows = array2D.GetLength(0);
        int columns = array2D.GetLength(1);
        for (int i = 0; i < rows; i++){
            for (int j = 0; j < columns; j++){
                Console.Write(array2D[i, j] + " ");
            }
            Console.WriteLine(); 
        }
    }
}
